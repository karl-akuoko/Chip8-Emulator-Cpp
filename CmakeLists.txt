# Minmum version of Cmake
cmake_minimum_required(VERSION 3.15)

# Project Name 
project(Chip8Emulator LANGUAGES CXX)

# Force C++17 for modern features
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Handle Homebrew non standard paths for SDL2 on Apple Silicon
if (APPLE)
    list(APPEND CMAKE_PREFIX_PATH "/opt/homebrew")
endif()

# Find the SDL2 Library
find_package(SDL2 REQUIRED)

# Include header files folder
include_directories(include)

# Source files
add_executable(${PROJECT_NAME}
    src/main.cpp
    src/cpu.cpp
)

# Help target SDL2 files 
target_include_directories(${PROJECT_NAME} PRIVATE ${SDL2_INCLUDE_DIRS} include)
target_link_libraries(${PROJECT_NAME} PRIVATE ${SDL2_LIBRARIES})
